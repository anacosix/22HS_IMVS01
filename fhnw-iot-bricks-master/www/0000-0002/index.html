import ch.fhnw.imvs.bricks.*;

BASE_URL = "brick.li/";

// or proxy = MockProxy.fromConfig(BASE_URL);

ButtonBrick brick = ButtonBrick.connect(proxy, BRICK_ID);

// this brick
boolean pressed = brick.isPressed();

// all bricks
int battery = brick.getBatteryLevel();
