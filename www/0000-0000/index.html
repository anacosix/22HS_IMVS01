Java Example

import java.awt.Color;
import ch.fhnw.imvs.bricks.*;

...

BASE_URL = "brick.li/";
BRICK_ID = "0000-0000";

Proxy proxy = MqttProxy.fromConfig(BASE_URL);
// or proxy = MockProxy.fromConfig(BASE_URL);

LedBrick brick = LedBrick.connect(proxy, BRICK_ID);

// this brick
Color color = brick.getColor();
brick.setColor(Color.RED);

// all bricks
int battery = brick.getBatteryLevel();
String time = brick.getTimestampIsoUtc();
