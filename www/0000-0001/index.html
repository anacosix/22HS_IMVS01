Humidity & Temperature

Java Example

import ch.fhnw.imvs.bricks.*;

...

BASE_URL = "brick.li/";
BRICK_ID = "0000-0001";

Proxy proxy = MqttProxy.fromConfig(BASE_URL);
// or proxy = MockProxy.fromConfig(BASE_URL);

HumiTempBrick brick = HumiTempBrick.connect(proxy, BRICK_ID);

// this brick
double humi = brick.getHumidity();
double temp = brick.getTemperature();

// all bricks
int battery = brick.getBatteryLevel();
String time = brick.getTimestampIsoUtc();
